class Solution {
    public ListNode removeDuplicateNodes(ListNode head) {
        if (head == null) {
            return head;
        }
        ListNode index=head;
        HashSet<Object> set = new HashSet<>();
        set.add(head.val);
        while (head.next!=null){
            if (set.add(head.next.val)){
                head=head.next;
            }else {
                head.next=head.next.next;
            };
        }
        head.next=null;
        return index;
    }
}
