class Solution {
    public int findGCD(int[] nums) {
        int min = 1001, max = -1;
        for(int num : nums){
            if(num < min) min = num;
            if(num > max) max = num;
        }
        return gcd(min, max); 
    }

    int gcd(int num1, int num2) {
        while(num1 > 0) {
            int tmp = num2 % num1;
            num2 = num1;
            num1 = tmp;
        }
        return num2;
    }
}
