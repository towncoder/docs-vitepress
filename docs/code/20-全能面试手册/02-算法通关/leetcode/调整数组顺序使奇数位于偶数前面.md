class Solution {
    public int[] exchange(int[] nums) {
        int l = 0, r = nums.length - 1;
        while (l < r){ 
            if(nums[l] % 2 == 0 && nums[r] % 2 == 1)
                swap(nums,l++,r--);
            if(nums[l] % 2 == 1) l++;
            if(nums[r] % 2 == 0) r--;
        }
        return nums;
    }
    private void swap(int[] nums, int index1, int index2){
        int temp = nums[index1];
        nums[index1] = nums[index2];
        nums[index2] = temp;
    }
}
