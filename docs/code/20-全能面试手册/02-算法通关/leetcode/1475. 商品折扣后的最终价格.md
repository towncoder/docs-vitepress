class Solution {
    public int[] finalPrices(int[] prices) {
        int n = prices.length;
        Stack<Integer> s = new Stack<>();
        for(int i=0; i<n; i++){
            while(! s.isEmpty() && prices[i] <= prices[s.peek()]){  //4<=8,2<=6
                prices[s.pop()] -= prices[i];   //8-4,6-2,4-2
            }
            s.push(i);  //3
        }
        return prices;
    }
}
