class Solution {
    public int maxProductDifference(int[] nums) {
        int n = nums.length;
        for(int i=0;i<n-1;i++){
            int index = i;
            for(int j=i+1;j<n;j++){
                if(nums[index] < nums[j])
                index = j;
            }
            int temp = nums[index];
            nums[index] = nums[i];
            nums[i] = temp;
        }
        int result = nums[0]*nums[1] - nums[n-1]*nums[n-2];
        return result;
    }
}
