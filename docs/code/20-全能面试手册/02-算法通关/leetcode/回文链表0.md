class Solution {
    public boolean isPalindrome(ListNode head) {
        Stack<Integer> stack = new Stack<>();
        ListNode l1=head;
        ListNode l2=head;
        while (l1!=null){
            stack.push(l1.val);
            l1=l1.next;
        }
        while (!stack.isEmpty()){
            if(stack.pop()!=l2.val){
                return false;
            }
            l2=l2.next;
        }
        return true;
    }
}
